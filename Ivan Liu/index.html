G <- gvisGeoChart(Exports, "Country", "Profit",options=list(width=200, height=100))
T <- gvisTable(Exports,options=list(width=200, height=270))
M <- gvisMotionChart(Fruits, "Fruit", "Year",options=list(width=400, height=370))

plot(GTM)
